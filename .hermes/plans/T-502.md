### Task 2: Create GraphView component

**File:** `packages/web/src/components/graph/GraphView.tsx` (CREATE)

```tsx
import { useState, useMemo, useCallback } from "react";
import {
  ReactFlow,
  ReactFlowProvider,
  Background,
  Controls,
  Minimap,
  useOnViewportChange,
  type Node,
  type NodeTypes,
  type EdgeTypes,
} from "@xyflow/react";
import "@xyflow/react/dist/style.css";
import { validateGraph } from "../../graph/validateGraph";
import { layoutGraph } from "../../graph/layoutGraph";
import { toReactFlow } from "../../graph/toReactFlow";
import { DocumentoNode } from "./DocumentoNode";
import { FimCadeiaNode } from "./FimCadeiaNode";
import { OrigemEdge } from "./OrigemEdge";
import type { DocumentoData } from "../../graph/types";
import "./GraphView.css";

const nodeTypes: NodeTypes = {
  documento: DocumentoNode,
  fimCadeia: FimCadeiaNode,
};

const edgeTypes: EdgeTypes = {
  origem: OrigemEdge,
};

interface GraphViewProps {
  graph: unknown;
}

export function GraphView({ graph }: GraphViewProps) {
  const [selectedNode, setSelectedNode] = useState<Node | null>(null);

  const { nodes, edges } = useMemo(() => {
    const validated = validateGraph(graph);
    const layouted = layoutGraph(validated);
    return toReactFlow(layouted);
  }, [graph]);

  const onNodeClick = useCallback((_event: React.MouseEvent, node: Node) => {
    setSelectedNode(node);
  }, []);

  const onPaneClick = useCallback(() => {
    setSelectedNode(null);
  }, []);

  const closePanel = useCallback(() => {
    setSelectedNode(null);
  }, []);

  const renderDetailContent = () => {
    if (!selectedNode) return null;
    const data = selectedNode.data;
    const isFim = selectedNode.type === "fimCadeia";
    return (
      <>
        <h3>{selectedNode.data.label || selectedNode.id}</h3>
        <div className="graph-view__panel-section">
          <div className="graph-view__panel-field">
            <span className="graph-view__panel-label">ID</span>
            <span className="graph-view__panel-value">{selectedNode.id}</span>
          </div>
          <div className="graph-view__panel-field">
            <span className="graph-view__panel-label">Type</span>
            <span className="graph-view__panel-value">{selectedNode.type}</span>
          </div>
          {isFim ? (
            <>
              <div className="graph-view__panel-field">
                <span className="graph-view__panel-label">Classificação</span>
                <span className="graph-view__panel-value value--{data.classificacao}">{data.classificacao}</span>
              </div>
            </>
          ) : (
            <>
              <div className="graph-view__panel-field">
                <span className="graph-view__panel-label">Número</span>
                <span className="graph-view__panel-value">{data.numero}</span>
              </div>
              <div className="graph-view__panel-field">
                <span className="graph-view__panel-label">Tipo</span>
                <span className="graph-view__panel-value">{data.tipo}</span>
              </div>
              <div className="graph-view__panel-field">
                <span className="graph-view__panel-label">Cartório</span>
                <span className="graph-view__panel-value">{data.cartorioId}</span>
              </div>
              <div className="graph-view__panel-field">
                <span className="graph-view__panel-label">Data</span>
                <span className="graph-view__panel-value">{data.data}</span>
              </div>
            </>
          )}
        </div>
      </>
    );
  };

  return (
    <div className="graph-view" data-testid="graph-view">
      <div className="graph-view__flow">
        <ReactFlowProvider>
          <ReactFlow
            nodes={nodes}
            edges={edges}
            nodeTypes={nodeTypes}
            edgeTypes={edgeTypes}
            fitView
            nodesDraggable={false}
            nodesConnectable={false}
            elementsSelectable={true}
            onNodeClick={onNodeClick}
            onPaneClick={onPaneClick}
            defaultEdgeOptions={{ animated: false }}
          >
            <Background />
            <Controls />
            <Minimap />
          </ReactFlow>
        </ReactFlowProvider>
      </div>
      <aside
        className={`graph-view__panel ${selectedNode ? "graph-view__panel--open" : "graph-view__panel--closed"}`}
        data-testid="detail-panel"
      >
        <div className="graph-view__panel-header">
          <h2>Detalhes</h2>
          <button
            className="graph-view__panel-close"
            onClick={closePanel}
            data-testid="detail-panel-close"
          >
            ✕
          </button>
        </div>
        <div className="graph-view__panel-body">
          {selectedNode ? renderDetailContent() : <p>Click em um nó para ver detalhes</p>}
        </div>
      </aside>
    </div>
  );
}
```

### Task 3: Rewrite graph route

**File:** `packages/web/src/routes/graph.tsx` (REWRITE)

```tsx
import { useState, useMemo } from "react";
import { GraphView } from "../components/graph/GraphView";
import { generateMockGraph, type MockShape } from "../graph/mock";

const SHAPES: Array<{ value: MockShape; label: string }> = [
  { value: "linear", label: "Linear" },
  { value: "branching", label: "Ramificado" },
  { value: "merge", label: "Fusão" },
  { value: "complex", label: "Complexo (20+ nós)" },
];

const GraphRoute = () => {
  const [shape, setShape] = useState<MockShape>("complex");
  const graph = useMemo(() => generateMockGraph(shape), [shape]);

  return (
    <div className="graph-page">
      <div className="graph-page__toolbar">
        <label htmlFor="mock-shape-select">Cadeia:</label>
        <select
          id="mock-shape-select"
          value={shape}
          onChange={(e) => setShape(e.target.value as MockShape)}
          data-testid="mock-shape-select"
        >
          {SHAPES.map((s) => (
            <option key={s.value} value={s.value}>
              {s.label}
            </option>
          ))}
        </select>
      </div>
      <GraphView graph={graph} />
    </div>
  );
};

export default GraphRoute;
```

test("graph page loads with complex mock by default", async ({ page }) => {

test("mock shape selector switches between shapes", async ({ page }) => {
  await page.route("**/health", async (route) => {
    await route.fulfill({ status: 200, contentType: "application/json", body: JSON.stringify({ ok: true }) });
  });
  await page.goto("/graph");

  const select = page.getByTestId("mock-shape-select");

  // Default is complex
  await expect(select).toHaveValue("complex");

  // Switch to linear → fewer nodes (6 = 5 doc + 1 fim)
  await select.selectOption("linear");
  await expect(page.locator(".react-flow__node")).toHaveCount(6);

  // Switch to branching (6 = 4 doc + 2 fim)
  await select.selectOption("branching");

  // Switch to merge (4 = 3 doc + 1 fim)
  await select.selectOption("merge");
